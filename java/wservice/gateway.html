路由 ​

四个属性含义如下：

  • id：路由的唯一标示

  • predicates：路由断言，其实就是匹配条件

  • filters：路由过滤条件，后面讲

  • uri：路由目标地址，lb://代表负载均衡，从注册中心获取目标微服务的实例列表，并且负载均衡选择一个访问。

    配置文件 ​

yaml
server:
  port: 8080
spring:
  application:
    name: gateway
  cloud:
    nacos:
      server-addr: 192.168.150.101:8848
    gateway:
      routes:
        - id: item # 路由规则id，自定义，唯一
          uri: lb://item-service # 路由的目标服务，lb代表负载均衡，会从注册中心拉取服务列表
          predicates: # 路由断言，判断当前请求是否符合当前规则，符合则路由到目标服务
            - Path=/items/**,/search/** # 这里是以请求路径作为判断规则
        - id: cart
          uri: lb://cart-service
          predicates:
            - Path=/carts/**
        - id: user
          uri: lb://user-service
          predicates:
            - Path=/users/**,/addresses/**
        - id: trade
          uri: lb://trade-service
          predicates:
            - Path=/orders/**
        - id: pay
          uri: lb://pay-service
          predicates:
            - Path=/pay-orders/**

网关登录校验 ​

java
@Component
@RequiredArgsConstructor
public class AuthGlobalFilter implements GlobalFilter, Ordered {

    private final JwtTool jwtTool;
    private final AuthProperties authProperties;
    private final AntPathMatcher antPathMatcher = new AntPathMatcher();

    /**
     * 实现GatewayFilter接口的filter方法，用于在网关路由之前进行权限验证。
     * 检查每个请求是否包含有效的用户令牌，如果不包含或令牌无效，则拒绝访问。
     *
     * @param exchange 服务器web交换对象，包含请求和响应信息。
     * @param chain    网关过滤器链，用于继续或中断请求处理链。
     * @return Mono<Void> 表示异步处理结果，用于表示响应的完成。
     */
    @Override
    public Mono<Void> filter(ServerWebExchange exchange, GatewayFilterChain chain) {
        // 获取请求对象
        ServerHttpRequest request = exchange.getRequest();
        // 检查当前请求路径是否在排除列表中，不在列表中则进行令牌验证
        if (isNeedYZ(request.getPath().toString())) {
            return chain.filter(exchange);
        }
        // 初始化用户ID为null，用于后续判断
        Long userId = null;
        // 初始化令牌变量
        String token = null;
        // 尝试从请求头中获取"Authorization"字段，用于获取令牌
        List<String> headers = request.getHeaders().get("Authorization");
        if (headers != null && headers.size() > 0) {
            token = headers.get(0);
        }

        // 尝试解析令牌，获取用户ID
        try {
            userId = jwtTool.parseToken(token);
        } catch (Exception e) {
            // 解析失败，表示令牌无效，设置响应状态码为未授权，并结束响应
            ServerHttpResponse response = exchange.getResponse();
            response.setStatusCode(HttpStatus.UNAUTHORIZED);
            return response.setComplete();
        }
        // 打印用户ID，用于调试
        System.out.println("userId = " + userId);
        String userInfo = userId.toString();
        ServerWebExchange swe = exchange.mutate()
                .request(builder -> builder.header("user-info", userInfo))
                .build();

        // 如果请求通过了所有检查，继续处理请求链
        return chain.filter(swe);
    }


    private boolean isNeedYZ(String path) {
        for (String pathPattern : authProperties.getExcludePaths()) {
            if (antPathMatcher.match(pathPattern, path)) {
                return true;
            }
        }
        return false;
    }


    @Override
    public int getOrder() {
        return 0;
    }
}
